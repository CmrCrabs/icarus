  (message "Emacs loaded in %s with %d garbage collections."
           (format "%.2f seconds"
                   (float-time
                    (time-subtract after-init-time before-init-time)))
           gcs-done))

(add-hook 'emacs-startup-hook #'efs/display-startup-time)

 site-run-file nil
 inhibit-default-init t
 gc-cons-threshold most-positive-fixnum)

          #'(lambda () (setq gc-cons-threshold (* 8 1024 1024))))

  (with-eval-after-load 'perspective
    (defun +tabs-buffer-list ()
      "overriding default buffer list to only include current perspective's buffers"
      (seq-filter
       (lambda (b)
         (and (buffer-live-p b)
              (not (string-prefix-p "*" (buffer-name b)))
              (not (string-prefix-p " " (buffer-name b)))))
       (persp-buffers (persp-curr))))

    (setq centaur-tabs-buffer-list-function #'+tabs-buffer-list)

    (defun centaur-tabs-buffer-groups ()
      "overrriding default buffer group to prevent any filetype/context grouping"
      (list "all"))))
