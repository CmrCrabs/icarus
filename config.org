        centaur-tabs-modified-marker "*"))

(with-eval-after-load 'centaur-tabs
  (with-eval-after-load 'perspective
    (defun +tabs-buffer-list ()
      "Overriding default buffer list to only include current perspective's buffers."
      (seq-filter
       (lambda (b)
         (and (buffer-live-p b)
              (not (string-prefix-p "*" (buffer-name b)))
              (not (string-prefix-p " " (buffer-name b)))))
       (persp-buffers (persp-curr))))

    (setq centaur-tabs-buffer-list-function #'+tabs-buffer-list)

    (defun centaur-tabs-buffer-groups ()
      "Overrriding default buffer group to prevent any filetype/context grouping."
      (list "all"))))
